    4、numpy矩阵操作与线性方程组

    from numpy.linalg import *

    print(np.eye(3)) # 生成单位矩阵

    [[ 1.  0.  0.]
 [ 0.  1.  0.]
 [ 0.  0.  1.]]

    list = np.array([[1.,2.],
               [3.,4.]])

    print(inv(list)) # 逆矩阵

    [[-2.   1. ]
 [ 1.5 -0.5]]

    print(list.transpose()) # 转置矩阵

    [[ 1.  3.]
 [ 2.  4.]]

     print(det(list)) # 行列式

    -2.0

    print(eig(list)) # 特征值与特征向量

    (array([-0.37228132,  5.37228132]), array([[-0.82456484, -0.41597356],
       [ 0.56576746, -0.90937671]]))

    y = np.array([[5.],[7.]]) 

    print(solve(list,y)) # 求解线性方程组

    [[-3.]
 [ 4.]]
